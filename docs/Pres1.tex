  \begin{frame}[fragile]{Persistent IV}
    \begin{minted}{haskell}
      dbRestAPI = toServant RestAPI { .. }
        where
          _getAll' = runDB (DB.selectList [] [])
          _addNew' res = do
              idx <- runDB (DB.insert res)
              pure $ Entity idx res
          _getSingle' idx =
              Entity idx
                <$> whenNothingM (runDB (DB.get idx)) 
                                 (throwError err404)
          _replace' idx res = do
              runDB (DB.repsert idx res)
              pure $ Entity idx res
          _delete' key = do
              runDB (DB.delete key)
              pure NoContent
    \end{minted}
